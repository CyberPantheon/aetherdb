- course_name (varchar(100)) NOT NULL

Table enrollments:
- enrollment_id (int(11)) NOT NULL  PRIMARY KEY
- student_id (int(11)) NOT NULL
- course_id (int(11)) NOT NULL
- grade (varchar(2)) NULL
- semester (varchar(10)) NULL
  Foreign Keys:

    - course_id → courses.course_id
  Rows: 8

Table students:
